131  //--------------------------------------------------------------------
132  void prepare()
133  {
134  double scale_x;
135  double scale_y;
136 
137  base_type::interpolator().transformer().scaling_abs(&scale_x, &scale_y);
138 
139  double scale_xy = scale_x * scale_y;
140  if (scale_xy > m_scale_limit)
141  {
142  scale_x = scale_x * m_scale_limit / scale_xy;
143  scale_y = scale_y * m_scale_limit / scale_xy;
144  }
145 
146  if(scale_x < 1) scale_x = 1;
147  if(scale_y < 1) scale_y = 1;
148 
149  if(scale_x > m_scale_limit) scale_x = m_scale_limit;
150  if(scale_y > m_scale_limit) scale_y = m_scale_limit;
151 
152  scale_x *= m_blur_x;
153  scale_y *= m_blur_y;
154 
155  if(scale_x < 1) scale_x = 1;
156  if(scale_y < 1) scale_y = 1;
157 
158  m_rx = uround( scale_x * double(image_subpixel_scale));
159  m_rx_inv = uround(1.0/scale_x * double(image_subpixel_scale));
160 
161  m_ry = uround( scale_y * double(image_subpixel_scale));
162  m_ry_inv = uround(1.0/scale_y * double(image_subpixel_scale));
163  }
164 
165  protected:
166  int m_rx;
167  int m_ry;
168  int m_rx_inv;
169  int m_ry_inv;
170 
171  private:
172  double m_scale_limit;
173  double m_blur_x;
174  double m_blur_y;
175  };
