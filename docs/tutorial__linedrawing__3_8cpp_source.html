105  agg::conv_dash<agg::conv_curve<CmdVertexPath>> dashPath (curvePath);
106 
107  // a long dash

109  // a dot

111 
112  // and then use this curved path to generate the stroke path
113  agg::conv_stroke<agg::conv_dash<agg::conv_curve<CmdVertexPath>>> strokePath (dashPath);
114 
115  strokePath.width(strokeWidth);
