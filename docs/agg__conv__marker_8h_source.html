87  template<class MarkerLocator, class MarkerShapes>
88  unsigned conv_marker<MarkerLocator, MarkerShapes>::vertex(double* x, double* y)
89  {
90  unsigned cmd = path_cmd_move_to;
91  double x1, y1, x2, y2;
92 
93  while(!is_stop(cmd))
94  {
95  switch(m_status)
96  {
97  case initial:
98  if(m_num_markers == 0)
99  {
100  cmd = path_cmd_stop;
101  break;
102  }
103  m_marker_locator->rewind(m_marker);
104  ++m_marker;
105  m_num_markers = 0;
106  m_status = markers;
107 
108  case markers:
109  if(is_stop(m_marker_locator->vertex(&x1, &y1)))
110  {
111  m_status = initial;
112  break;
113  }
114  if(is_stop(m_marker_locator->vertex(&x2, &y2)))
115  {
116  m_status = initial;
117  break;
118  }
119  ++m_num_markers;
120  m_mtx = m_transform;
121  m_mtx *= trans_affine_rotation(std::atan2(y2 - y1, x2 - x1));
122  m_mtx *= trans_affine_translation(x1, y1);
123  m_marker_shapes->rewind(m_marker - 1);
124  m_status = polygon;
125 
126  case polygon:
127  cmd = m_marker_shapes->vertex(x, y);
128  if(is_stop(cmd))
129  {
130  cmd = path_cmd_move_to;
131  m_status = markers;
132  break;
133  }
134  m_mtx.transform(x, y);
135  return cmd;
136 
137  case stop:
138  cmd = path_cmd_stop;
139  break;
140  }
141  }
142  return cmd;
143  }
